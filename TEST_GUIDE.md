```bash

python -m pytest tests/unit/ -v

# Integration tests
python -m pytest tests/integration/ -v -m integration

# Scenario tests
python -m pytest tests/scenarios/ -v -m scenario

# Full pyramid (run all in order)
python -m pytest tests/unit/ -v && \
  python -m pytest tests/integration/ -v -m integration && \
  python -m pytest tests/scenarios/ -v -m scenario
```
